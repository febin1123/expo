                                                                          public static boolean RatMaze(RatPair[][] a, int rowi, int coli, int maxrow, int maxcol) {
                                                                          if (rowi == maxrow - 1 && coli == maxcol - 1) {
                                                                          a[rowi][coli].isvisited = true;
                                                                          return true;
                                                                          }

                                                                          // forward
                                                                          if (rowi + 1 <= maxrow && coli + 1 <= maxcol && a[rowi][coli].data == 1) {
                                                                          boolean b = RatMaze(a, rowi, coli + 1, maxrow, maxcol);
                                                                          if (b) {
                                                                          a[rowi][coli].isvisited = true;
                                                                          }
                                                                          }

                                                                          if (rowi + 1 <= maxrow && coli + 1 <= maxcol && a[rowi][coli].data == 1) {

                                                                          boolean b = RatMaze(a, rowi + 1, coli, maxrow, maxcol);
                                                                          if (b) {
                                                                          a[rowi][coli].isvisited = true;
                                                                          }
                                                                          }

                                                                          if (rowi + 1 <= maxrow && coli + 1 <= maxcol)
                                                                          return a[rowi][coli].isvisited;
                                                                          else {
                                                                          return false;
                                                                          }
                                                                          }

                                                                          }

                                                                          int data = 1;
                                                                          boolean isvisited;
                                                                          }

                                                                                        Simple c++ code for backtracking using recursion

                                                                                        #include <iostream>
                                                                                        using namespace std;
                                                                                        int x=0, y=0;
                                                                                        void backTracking(int maze[][4],int sol[][4])
                                                                                        {
                                                                                        if (maze[x][y] == 2)
                                                                                        {
                                                                                        sol[x][y] = 1;
                                                                                        cout << "PATH FOUND HURRAH!\n";
                                                                                        }
                                                                                        else
                                                                                        {
                                                                                        if (maze[x][y + 1] != 0)
                                                                                        {
                                                                                        sol[x][y] = 1;
                                                                                        y++;
                                                                                        backTracking(maze,sol);
                                                                                        }
                                                                                        else
                                                                                        {
                                                                                        if (maze[x + 1][y] == 1)
                                                                                        {
                                                                                        sol[x][y] = 1;
                                                                                        x++;
                                                                                        backTracking(maze,sol);
                                                                                        }
                                                                                        }
                                                                                        }
                                                                                        }
